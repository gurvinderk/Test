pipeline {
    agent any
    stages{
        stage ("compile stage"){
            steps {
                    bat 'mvn clean compile'
            }
        }

        stage ("Testing stage"){
            steps {
                    bat 'mvn test'
            }
        }
        stage ("Deployment stage"){
            steps {
                    bat 'mvn  deploy'
            }
        }
    }
}
